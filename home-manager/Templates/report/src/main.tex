\perlnewcommand\includefile[2]{
    my ($title, $name) = @_;
    return "\\section{$title}\n\\input{src/$name}";
}

\newcommand{\figureRef}[1]{ \ref{fig:#1} }

\perlnewcommand\figureImage[2]{
	my ($name, $caption) = @_;

	return sprintf '\
	\begin{figure}[!htb]
	\includegraphics[width=0.5\textwidth]{%s}
	\caption{%s}
	\label{fig:%s}
	\end{figure}', $name, $caption, $name;
}
